```python
"""Tidy data exports for R/ggplot figures (compute here, render in R)."""

import pandas as pd

_LOCAL_UTC_OFFSET = pd.Timedelta(hours=10)  # AEST, as everywhere else


def hotspots_for_days(hotspots: pd.DataFrame, days) -> pd.DataFrame:
    """Hotspots whose AEST local day is in `days` -> date, lat, lon, frp."""
    local_day = (hotspots["datetime_utc"].dt.tz_localize(None) + _LOCAL_UTC_OFFSET).dt.normalize()
    wanted = pd.to_datetime(pd.Index(days))
    keep = local_day.isin(wanted)
    out = hotspots.loc[keep, ["lat", "lon", "frp"]].copy()
    out.insert(0, "date", local_day[keep].values)
    return out.reset_index(drop=True)


def benchmark_table(panel: pd.DataFrame, benchmarks: dict) -> pd.DataFrame:
    """Per benchmark event: DLI, tier, and within-tier percentile on the peak day."""
    p = panel.set_index("date")
    rows = []
    for name, day in benchmarks.items():
        row = p.loc[pd.Timestamp(day)]
        tier = int(row["confidence_tier"])
        in_tier = p[p["confidence_tier"] == tier]["dli"]
        rows.append({
            "name": name,
            "date": pd.Timestamp(day),
            "dli": float(row["dli"]),
            "confidence_tier": tier,
            "pct": float((in_tier < row["dli"]).mean()),
        })
    return pd.DataFrame(rows)
```

```python
"""Count DRFA activations per LGA and check the name-join against ABS boundaries."""

import sys
from pathlib import Path

import geopandas as gpd
import pandas as pd

sys.path.insert(0, str(Path(__file__).resolve().parents[1]))

from scripts.config import DATA_DERIVED, PATHS

EXPORT = DATA_DERIVED.parent / "export"

raw = pd.read_csv(PATHS.drfa_locations)
# locate the LGA-name column robustly (exact name varies between exports)
lga_col = next(c for c in raw.columns if "lga" in c.lower() or "local government" in c.lower())
counts = (raw.groupby(raw[lga_col].str.strip().str.upper())
             .size().rename("n_activations").reset_index()
             .rename(columns={lga_col: "lga_name"}))

shp = next(Path("data/raw/abs_lga").glob("*.shp"), None) or \
      next(Path("data/raw/abs_lga").glob("*.geojson"))
bounds = gpd.read_file(shp)
name_col = next(c for c in bounds.columns if "NAME" in c.upper())
bounds["lga_name"] = bounds[name_col].str.strip().str.upper()

matched = counts["lga_name"].isin(bounds["lga_name"])
print(f"name join: {matched.mean():.1%} of {len(counts)} DRFA LGAs matched", flush=True)
print("unmatched examples:", counts.loc[~matched, "lga_name"].head(10).tolist(), flush=True)

counts.to_csv(EXPORT / "fig_drfa_lga.csv", index=False)
bounds[["lga_name", "geometry"]].to_file(EXPORT / "lga_boundaries.geojson", driver="GeoJSON")
```
